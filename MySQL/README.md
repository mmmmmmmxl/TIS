#常用命令#
```
#创建数据库
create database 数据库名

#显示数据库列表
show databases;

#使用其中某个数据库
use database_name;

#查看某个数据库中所有的表
show table status from your_table_name;

#查看某个表的所有字段
select COLUMN_NAME from information_schema.COLUMNS where table_name = 'your_table_name';

#查看某个表的所有值
select * from your_table_name;

```
